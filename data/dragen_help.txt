DRAGEN Host Software Version 07.021.604.3.7.8

Run Map/Align and Variant Caller (*.fastq to *.vcf)
  dragen -r <REF_DIR> --output-directory <OUT_DIR> --output-file-prefix <FILE_PREFIX> [options]
  -1 <FASTQ> [-2 <FASTQ>] --RGID <RG> --RGSM <SM> --enable-variant-caller true

Run Map/Align (*.fastq to *.bam)
  dragen -r <REF_DIR> --output-directory <OUT_DIR> --output-file-prefix <FILE_PREFIX> [options]
  -1 <FASTQ> [-2 <FASTQ>] --RGID <RG> --RGSM <SM>

Run Variant Caller (*.bam to *.vcf)
  dragen -r <REF_DIR> --output-directory <OUT_DIR> --output-file-prefix <FILE_PREFIX> [options]
  -b <BAM> --enable-variant-caller true

Run RNA-Seq Map/Align (*.fastq to *.bam)
  dragen -r <REF_DIR> --output-directory <OUT_DIR> --output-file-prefix <FILE_PREFIX> [options]
  -1 <FASTQ> [-2 <FASTQ>] --RGID <RG> --RGSM <SM> --enable-rna true

Run BCL Conversion (BCL directory to *.fastq.gz)
  dragen --bcl-conversion-only true --bcl-input-directory <BCL_ROOT_DIR> --output-directory <PATH> [options]

Build Hash Table
  dragen --build-hash-table true --output-directory <OUT_DIR> --ht-reference <REF_FASTA> [options]
